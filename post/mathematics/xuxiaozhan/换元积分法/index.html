换元法类型

所用的凑微分公式

\(\displaystyle \int{f(ax + b)dx = \frac{1}{a}}\int{f(ax + b)d(ax + b)}\)

\(\displaystyle dx = \frac{1}{a}d(ax + b)\)

\(\displaystyle \int{f(x^n)x^{n-1}dx = \frac{1}{n}}\int{f(x^n)dx^n}\)

\(\displaystyle x^{n-1}dx = \frac{1}{n}dx^n\)

\(\displaystyle \int{f(x^2)xdx = \frac{1}{2}}\int{f(x^2)dx^2}\)

\(\displaystyle xdx = \frac{1}{2}dx^2\)

\(\displaystyle \int{\frac{f(\sqrt{x})}{\sqrt{x}}dx} = 2\int{f(\sqrt{x})d\sqrt{x}}\)

\(\displaystyle \frac{1}{\sqrt{x}}dx = 2d\sqrt{x}\)

\(\displaystyle \int{f(\frac{1}{x})\frac{1}{x^2}dx} = -\int{f(\frac{1}{x})d\frac{1}{x}}\)

\(\displaystyle \frac{1}{x^2}dx = -d\frac{1}{x}\)

\(\displaystyle \int{\frac{f(\ln{x})}{x}dx} = \int{f(\ln{x})d\ln{x}}\)

\(\displaystyle \frac{1}{x}dx = d\ln{x}\)

\(\displaystyle \int{f(e^x)e^xdx} = \int{f(e^x)de^x}\)

\(\displaystyle e^xdx = de^x\)

\(\displaystyle \int{f(\sin{x})\cos{x}dx} = \int{f(\sin{x})d\sin{x}}\)

\(\displaystyle \cos{x}dx = d\sin{x}\)

\(\displaystyle \int{f(\cos{x})\sin{x}dx} = -\int{f(\cos{x})d\cos{x}}\)

\(\displaystyle \sin{x}dx = -d\cos{x}\)

\(\displaystyle \int{\frac{f'(x)}{f(x)}dx} = \int{\frac{1}{f(x)d(f(x))}} = \ln{|f(x)| + C}\)

\(\displaystyle f'(x)dx = df(x)\)

\(\displaystyle \int{\frac{f(\arctan{x})}{1 + x^2}dx} = -\int{f(\arctan{x})d\arctan{x}}\)

\(\displaystyle \frac{1}{1+x^2}dx = d\arctan{x}\)

\(\displaystyle \int{\frac{f(\arcsin{x})}{\sqrt{1 - x^2}}dx} = -\int{f(\arcsin{x})d\arcsin{x}}\)

\(\displaystyle \frac{1}{\sqrt{1 - x^2}}dx = d\arcsin{x}\)

\(\displaystyle \int{\frac{f(\tan{x})}{\cos^2{x}}dx} = \int{f(\tan{x})\sec^2{x}dx} = \int{f(\tan{x})d\tan{x}}\)

\(\displaystyle \frac{1}{\cos^2{x}}dx = \sec^2{x}dx = d\tan{x}\)

\(\displaystyle \int{\frac{f(\cos{x})}{\sin^2{x}}dx} = \int{f(\cot{x})\csc^2{x}dx} = -\int{f(\cos{x})d\cot{x}}\)

\(\displaystyle \frac{1}{\sin^2{x}}dx = \csc^2{x}dx = -d\cot{x}\)
