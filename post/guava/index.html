/**
*	价格排序
*/
private void sortByPrice(List<Mlq_goods> mg_l,final int i) {  // 0:升序 1:降序
        Ordering<Mlq_goods> byPrice = new Ordering<Mlq_goods>() {
            @Override
            public int compare(Mlq_goods mg1, Mlq_goods mg2) {
                if (ObjUtils.notEmpty(mg1.getMp_l()) && ObjUtils.notEmpty(mg2.getMp_l())) {

                    String product_price1 = mg1.getMp_l().get(0).getProduct_price();
                    String product_price2 = mg2.getMp_l().get(0).getProduct_price();

                    Double price1 = Double.parseDouble(product_price1==null?"0":product_price1);
                    Double price2 = Double.parseDouble(product_price2==null?"0":product_price2);
                    if(0==i){
                        return price1> price2? 1:(price1 == price2 ? 0:-1);
                    }
                    return price1< price2? 1:(price1 == price2 ? 0:-1);
                }

                return 0;
            }
        };
        Collections.sort(mg_l,byPrice);
    }
