1. Jrebel 热部署

1.1. 安装

  1. 下载地址
    http://196.168.1.33:9080/f/d68fbbf872/

  2. 解压 jrebel_6.2.2.zip 到 tomcat 安装目录

  3. 修改 jrebel_6.2.2 中的 jrebel.properties

rebel.dirs=/apache-tomcat-7.0.62/webapps/app/WEB-INF/classes/ (1)
1 修改 classes 编译目录

1.2. 脚本

在 tomcat/bin 目录下创建文件

windows startup_jrebel_jvm.bat
@echo off
rem ---------------------------放入tomcat路径下的bin/catalina.bat同一级目录
rem ---------------------------jrebel remoting------------------------------------------------
cd ..
set "CATALINA_HOME=%cd%"
set JAVA_OPTS=%JAVA_OPTS% -noverify -javaagent:%CATALINA_HOME%\jrebel_6.2.2\jrebel.jar -Drebel.properties=%CATALINA_HOME%\jrebel_6.2.2\jrebel.properties -Drebel.disable_update=true

rem ---------------------------JVM------------------------------------------------
REM set JAVA_OPTS=%JAVA_OPTS% -Xms512m -Xmx512m -XX:PermSize=256m -XX:MaxPermSize=256m

call "%CATALINA_HOME%\bin\catalina.bat" %* jpda run
linux startup_jrebel_jvm.sh 
#!/bin/bash

CATALINA_HOME=/apache-tomcat-7.0.62

export CATALINA_OPTS="$CATALINA_OPTS -noverify -javaagent:$CATALINA_HOME/jrebel_6.2.2/jrebel.jar -Drebel.properties=$CATALINA_HOME/jrebel_6.2.2/jrebel.properties"

$CATALINA_HOME/bin/catalina.sh jpda run
