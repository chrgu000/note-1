1.1. Docker 安装

  1. 配置 本地 yum 仓库

    yum install -y yum-plugin-downloadonly # yum下载软件包插件
mkdir -p /var/www/html/data/yum-repo/ # yum 仓库目录
yum install docker-engine-1.12.1-1.el7.centos --downloadonly --downloaddir=/var/www/html/data/yum-repo/ # 下载 docker 依赖到仓库
createrepo -p -d -o /var/www/html/data/yum-repo/ /var/www/html/data/yum-repo/ # 生成 yum 仓库信息

systemctl start httpd # 启动 apache (如果没有 yum install -y httpd 安装)

  2. 配置 内网 yum 源

    cat > /etc/yum.repos.d/CentOS-Base.repo <<_EOF_
[Local-rpm]
name=Local-rpm
baseurl=http://jzlh.com/data/yum-repo
enabled=1
gpgcheck=0
_EOF_

  3. 配置 远程 yum 源

    cat > /etc/yum.repos.d/docker-main.repo <<_EOF_
[docker-main-repo]
name=Docker main Repository
baseurl=http://mirrors.aliyun.com/docker-engine/yum/repo/main/centos/7
enabled=1
gpgcheck=1
gpgkey=http://mirrors.aliyun.com/docker-engine/yum/gpg
_EOF_

  4. 安装 docker

    yum install -y docker-engine-1.12.1-1.el7.centos

  5. 关闭防火墙

    systemctl status firewalld.service #查看firewall状态

systemctl stop firewalld.service #停止firewall

systemctl disable firewalld.service #停止firewall开机启动

  6. 启动

    systemctl enable docker # 开机启动
systemctl start docker # 启动

  7. 如何使用Docker加速器

    mkdir -p /etc/docker
tee /etc/docker/daemon.json <<-'EOF'
{
  "registry-mirrors": ["https://7xefeire.mirror.aliyuncs.com"]
}
EOF
systemctl daemon-reload
systemctl restart docker

1.2. Rancher 安装

1.2.1. registry server

  1. 生成自签名证书

    cd ~/

mkdir registry && cd registry && mkdir certs && cd certs

openssl req -x509 -days 3650 -subj '/CN=mailiqing.com/' -nodes -newkey rsa:2048 -keyout registry.key -out registry.crt
    CN=mailiqing.com 这里是域名

  2. 拉取 registry:2.5

    docker pull registry:2.5

  3. 启动 registry server

    registry.yml
    version: '2'

services:
  registry-v2:
    image: registry:2.5
    container_name: registry-v2
    ports:
      - "5000:5000"
    environment:
      REGISTRY_HTTP_TLS_CERTIFICATE: /certs/registry.crt
      REGISTRY_HTTP_TLS_KEY: /certs/registry.key
    volumes:
      - /home/wuliu/registry-data:/var/lib/registry
      - /home/wuliu/registry/certs:/certs
  registry-web: (1)
    image: konradkleine/docker-registry-frontend:v2
    container_name: registry-web
    ports:
      - "8082:80"
    environment:
      ENV_DOCKER_REGISTRY_HOST: jzlh.com
      ENV_DOCKER_REGISTRY_PORT: 5000
      ENV_DOCKER_REGISTRY_USE_SSL: 1
    extra_hosts:
      - "jzlh.com:192.168.1.55"
    1 web界面查看镜像列表 
    docker-compose -f registry.yml up

1.2.3. 下载 Rancher 所需镜像并 Push 到 Docker Registry

  1. 脚本

    push.sh
    #!/bin/bash

for var in $@
do
    echo "$var"
    docker pull $var
    docker tag $var mailiqing.com:5000/$var
    docker push mailiqing.com:5000/$var
    docker rmi mailiqing.com:5000/$var
done

  2. 运行

    sh push.sh rancher/net:v0.11.2 rancher/net:holder rancher/scheduler:v0.7.5 rancher/healthcheck:v0.2.3 rancher/metadata:v0.9.1 rancher/dns:v0.14.2 rancher/network-manager:v0.6.6 rancher/agent:v1.2.2 rancher/server:v1.5.5
