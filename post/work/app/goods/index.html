JSON

否

{

  "catId":"",  # 管理类目id

  "productName":"",  # 型号
  "region":"",  # 出库地
  "sellerNick":"",  # 供应商
  "priceFrom":"",  # 起始价格
  "priceTo":"", # 结束价格
  "category":"",  # 商品类目
  "shopId":"",  # 店铺id

}
