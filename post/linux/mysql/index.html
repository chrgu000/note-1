1.1. MySQL拼音首字母查询

英文26个字母中除了i,u,v三个字母不能成为汉语拼音的首字母外，其它的字母都可以。如果我们需要查询结果集中以A或者其它字母开头的汉语，实现方法如下

SELECT
  *
FROM
  mlq_shop_info
WHERE CONV(
    HEX(LEFT(CONVERT(shop_name USING gbk), 1)),
    16,
    10
  ) BETWEEN 45253
  AND 45760

  FROM

      msi2.id

4. 根据 某一列的值 if else

SELECT
    mpa.product_id,
      MAX(
        CASE
          mpa.attr_name
          WHEN '销售地区'
          THEN IF(INSTR(mpa.attr_value, '-'),tranAddr(mpa.attr_value, '-'),CAST(mpa.attr_value AS CHAR(50)))
          ELSE NULL
        END
      ) sale_region,
      MAX(
        CASE
          mpa.attr_name
          WHEN '出库地区'
          THEN IF(INSTR(mpa.attr_value, '-'),tranAddr(mpa.attr_value, '-'),CAST(mpa.attr_value AS CHAR(50)))
          ELSE NULL
        END
      ) repository_region,
      MAX(
        CASE
          mpa.attr_name
          WHEN '出库地区'
          THEN IF(INSTR(mpa.attr_value, '-'),tranAddr(mpa.attr_value, '-'),CAST(mpa.attr_value AS CHAR(50)))
          ELSE NULL
        END
      ) repository_region_noa,
    MAX(
      CASE
        mpa.attr_name
        WHEN '最小起订量'
        THEN mpa.attr_value
        ELSE 0
      END
    ) min_quantity,
    MAX(
      CASE
        mpa.attr_name
        WHEN '库存量'
        THEN mpa.attr_value
        ELSE 0
      END
    ) stock_quantity,
    MAX(
      CASE
        mpa.attr_name
        WHEN '单位'
        THEN mpa.attr_value
        ELSE '吨'
      END
    ) unit,
    MAX(
      CASE
        mpa.attr_name
        WHEN '交货天数'
        THEN mpa.attr_value
        ELSE 7
      END
    ) delivery_day,
    MAX(
      CASE
        mpa.attr_name
        WHEN '包装类型'
        THEN mpa.attr_value
        ELSE '散装'
      END
    ) package_type
  FROM
    mlq_product_attr mpa
  WHERE mpa.product_id = '201611111012491'
