1. Deploy a Replica Set

1.1. 安装 & 下载

  1. 3.2.8
    [[mongodb-linux-x86_64-rhel70-3.2.8.tgz|https://fastdl.mongodb.org/linux/mongodb-linux-x86_64-rhel70-3.2.8.tgz]]

  2. 创建软链接

ln -s /mongod/bin/mongod /usr/bin/mongod
ln -s /mongod/bin/mongo /usr/bin/mongo

  1. 创建数据存储路径

mkdir -p /srv/mongodb/rs0-0 /srv/mongodb/rs0-1 /srv/mongodb/rs0-2

  1. 配置文件

    1. rs1.conf

systemLog:
  destination: file
  path: "/var/log/mongodb/rs1.log" #日志路径
  logAppend: true
storage:
  dbPath: "/srv/mongodb/rs0-0"
processManagement: #后台运行
  fork: true
net:
  bindIp: 0.0.0.0 #绑定所有网卡
  port: 27017
replication:
   oplogSizeMB: 128
   replSetName: "rs0"

  1. rs2.conf

systemLog:
  destination: file
  path: "/var/log/mongodb/rs2.log" #日志路径
  logAppend: true
storage:
  dbPath: "/srv/mongodb/rs0-1"
processManagement: #后台运行
  fork: true
net:
  bindIp: 0.0.0.0 #绑定所有网卡
  port: 27018
replication:
   oplogSizeMB: 128
   replSetName: "rs0"

  1. rs3.conf

systemLog:
  destination: file
  path: "/var/log/mongodb/rs3.log" #日志路径
  logAppend: true
storage:
  dbPath: "/srv/mongodb/rs0-2"
processManagement: #后台运行
  fork: true
net:
  bindIp: 0.0.0.0 #绑定所有网卡
  port: 27019
replication:
   oplogSizeMB: 128
   replSetName: "rs0"

  1. 连接 mongod

mongo --port 27017

  1. 初始化

# hostname 替换为主机名
rsconf = {
           _id: "rs0",
           members: [
                      {
                       _id: 0,
                       host: "<hostname>:27017"
                      }
                    ]
         }
rs.initiate( rsconf )

  1. 查看副本配置

rs.conf()

  1. 添加副本

# hostname 替换为主机名
rs.add("<hostname>:27018")
rs.add("<hostname>:27019")
